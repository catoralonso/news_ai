from google.adk.agents import LlmAgent
from agents.orchestrator.agent import Orchestrator

# Inicializar el orchestrator real
orch = Orchestrator()
orch.build_agents()

# Exponer como tools ADK
def run_pipeline(query: str) -> str:
    """Researches trends, fact-checks, writes article and publishes."""
    return orch.run(query)

def chat_reader(message: str) -> str:
    """Handles reader questions and fact-check requests."""
    response = orch.chat_reader(message)
    return response.message

# Root agent que ADK necesita encontrar
root_agent = LlmAgent(
    name="newspaper_ai",
    model="gemini-2.0-flash",
    instruction=f"You are the orchestrator of Nutrición AI newspaper...",
    tools=[run_pipeline, chat_reader],
)
